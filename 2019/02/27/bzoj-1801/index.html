1801: [Ahoi2009]chess 中国象棋
Time Limit: 10 Sec Memory Limit: 64 MB

Description
在N行M列的棋盘上，放若干个炮可以是0个，使得没有任何一个炮可以攻击另一个炮。 请问有多少种放置方法，中国像棋中炮的行走方式大家应该很清楚吧.

Input
一行包含两个整数N，M，中间用空格分开.

Output
输出所有的方案数，由于值比较大，输出其mod 9999973

Sample Input
1 3

Sample Output
7

HINT
除了在3个格子中都放满炮的的情况外，其它的都可以
100%的数据中N,M不超过100
50%的数据中，N,M至少有一个数不超过8
30%的数据中，N,M均不超过6

Solution:
显然就是求每行每列不超过2个的数量
难也就难在状态表示
f[i][j][k]表示前i行有j列只放了1个，有k列放了2个的数量
转移有6种，看代码

Code:

#include<bits/stdc++.h>
using namespace std;
const int mod=9999973;
//const int mod=1e9+7;
typedef long long ll;
const int N=105;
const int M=N<<1;

#define INF 0x3f3f3f3f

#define pb push_back

#define clr(x) memset(x,0,sizeof(x))

#define mp make_pair
//const int dx[4]={1,0,-1,0};
//const int dy[4]={0,1,0,-1};
int T,n,kase,qry,ql,qr,op,m;
ll f[2][N][N];
inline ll C(int x){
return x*(x-1)/2;
}

int main(){
scanf(“%d%d”,&n,&m);
f[0][0][0]=1;
//f[1][0][0]=1;f[1][1][0]=m;f[1][2][0]=C(m);
/for (int i=1;i<=n;i++){
for (int j=0;j<=m;j++){
for (int k=0;k+j<=m;k++){
f[i][j][k]=f[i-1][j][k];
if (j) f[i][j][k]+=f[i-1][j-1][k](m-k-j+1);//放一个在j
if (k) f[i][j][k]+=f[i-1][j+1][k-1](j+1);//放一个在k
//if (k) f[i][j][k]+=f[i-1][j][k-1](m-j-k+1);//不能放两个在同一k…
if (k>1) f[i][j][k]+=f[i-1][j+2][k-2]C(j+2);
if (j>1) f[i][j][k]+=f[i-1][j-2][k]C(m-k-j+2);//放两个都在j
if (j && k) f[i][j][k]+=f[i-1][j][k-1](j(m-j-k+1));//1j 1k
f[i][j][k]%=mod;
}
}
}/
int cur;
for (int i=1;i<=n;i++){
cur=i&1;
for (int j=0;j<=m;j++){
for (int k=0;k+j<=m;k++){
ll &tmp=f[cur][j][k];
tmp=f[!cur][j][k];
if (j) tmp+=f[!cur][j-1][k](m-k-j+1);//放一个在j
if (k) tmp+=f[!cur][j+1][k-1](j+1);//放一个在k
//if (k) f[i][j][k]+=f[i-1][j][k-1](m-j-k+1);//不能放两个在同一k…
if (k>1) tmp+=f[!cur][j+2][k-2]C(j+2);
if (j>1) tmp+=f[!cur][j-2][k]C(m-k-j+2);//放两个都在j
if (j && k) tmp+=f[!cur][j][k-1](j(m-j-k+1));//1j 1k
f[cur][j][k]%=mod;
}
}
}
ll ans=0;
for (int j=0;j<=m;j++)
for(int k=0;k+j<=m;k++) (ans+=f[cur][j][k])%=mod;
printf(“%lld”,ans);
}
